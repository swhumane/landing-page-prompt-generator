<script>

function generatePrompt() {

let brandName = document.getElementById("brandName").value;

let mainServices = document.getElementById("mainServices").value;

let locationType = document.getElementById("locationType").value;

let locationName = document.getElementById("locationName").value;

let primaryKeyword = document.getElementById("primaryKeyword") ? document.getElementById("primaryKeyword").value : "";

let mainKeywords = document.getElementById("mainKeywords").value;

let supportingKeywords = document.getElementById("supportingKeywords").value;

let email = document.getElementById("email").value;

let contactNumber = document.getElementById("contactNumber").value;

let brandInfo = document.getElementById("brandInfo").value;

 

let extraLocationText = "";

if (locationType === "State" || locationType === "County") {

extraLocationText = ` Mention top 10 to 15 cities in the ${locationType.toLowerCase()} of ${locationName} and tell that we provide service there also and in all other cities and towns.`;

}

 

let prompt = `Write landing page content for ${brandName} which provides ${mainServices}.

Location: ${locationType} - ${locationName}.${extraLocationText}

Use the primary keyword "${primaryKeyword}" in the first and the last paragraph and in the H1 heading also.

Use the following keywords in the H2 headings as well as in the paragraph below that: ${mainKeywords}.

Use the following keywords only in the paragraph and not in any headings: ${supportingKeywords}.

Give meta title of 60 characters using the primary keyword.

Give meta description of 155 characters using the other keywords.

Word count: 800 words.

Use simple language.

Start the content with a question addressing the problem.

Give 4 to 5 CTA button texts.

Use email ${email} and contact number ${contactNumber} in the last CTA paragraph.

Write content that must answer semantic queries on search engines.

Specific brand information: ${brandInfo}

Based upon the above content, act like Googles People Also Ask (PAA) box.

For the content above, generate a structured list of related questions using a tree hierarchy format:

Each main question should be in H3 format (###), followed by a brief and informative answer.

Each main question should expand into 23 follow-up questions, using bulleted indentation to represent 23 levels of depth.

Each sub-question should also include a concise and helpful answer directly beneath it.

Go at least 2 levels deep, and ideally 3 levels deep where it makes sense.

Format of the result:

Meta title

Meta description

CTA button texts

1

2

3

4

5

Content

FAQs`;

 

document.getElementById("output").value = prompt;

}

 

function copyPrompt() {

let output = document.getElementById("output");

output.select();

document.execCommand("copy");

alert("Prompt copied to clipboard!");

}

</script>
